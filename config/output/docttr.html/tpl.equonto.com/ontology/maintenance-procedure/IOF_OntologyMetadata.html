    ottr:IRI ?ontologyIri,
    !? ottr:IRI ?ontologyVersionIri,
    !? xsd:string ?label,
    !? xsd:string ?license,
    !? xsd:string ?abstract,

    ottr:Triple(?ontologyIri, dcterms:abstract, ?abstract),
